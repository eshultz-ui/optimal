/* ---------------- Geometric distance matrix ---------------- */
// Great-circle distance in meters
function havMeters(a,b){
  const R=6371000, toR=x=>x*Math.PI/180;
  const dLa=toR(b.lat-a.lat), dLn=toR(b.lng-a.lng), la1=toR(a.lat), la2=toR(b.lat);
  const h=Math.sin(dLa/2)**2+Math.cos(la1)*Math.cos(la2)*Math.sin(dLn/2)**2;
  return 2*R*Math.asin(Math.sqrt(h));
}
// Built-in distance assumptions (no user input needed):
// straight-line distance × ROAD_FACTOR ≈ real street mileage; AVG_MPH sets drive time.
const ROAD_FACTOR=1.3, AVG_MPH=35;
// Build distance(m)+duration(s) matrices geometrically. No network, no stop limit.
// Async only to keep callers simple.
async function getMatrix(points, onMsg){
  const N=points.length;
  const factor=ROAD_FACTOR, mph=AVG_MPH, speed=mph*0.44704; // m/s
  onMsg && onMsg(N>200?'Building distance matrix…':'Measuring distances…');
  const dist=new Array(N), dur=new Array(N);
  for(let i=0;i<N;i++){ dist[i]=new Float64Array(N); dur[i]=new Float64Array(N); }
  for(let i=0;i<N;i++){
    for(let j=i+1;j<N;j++){
      const m=havMeters(points[i],points[j])*factor;
      dist[i][j]=dist[j][i]=m; dur[i][j]=dur[j][i]=m/speed;
    }
  }
  return { dist, dur, mode:'geometric', factor, mph, N };
}

/* ---------------- VRP solver ----------------
   nodes: 0 = base, 1..n = jobs.  Fixed K routes, each base->...->base.
   C = cost matrix for chosen objective. Reports use real miles + hours.
*/
function solveVRP(C, n, K, coords, balance){
  const jobs=[]; for(let i=1;i<=n;i++) jobs.push(i);
  const routeCost = seq => { let s=0,p=0; for(const v of seq){s+=C[p][v];p=v;} return s+C[p][0]; };

  function twoOpt(seq){
    let improved=true;
    while(improved){ improved=false;
      for(let i=0;i<seq.length-1;i++){
        for(let k=i+1;k<seq.length;k++){
          const a=(i===0?0:seq[i-1]), b=seq[i];
          const c=seq[k], d=(k===seq.length-1?0:seq[k+1]);
          if(a===c||b===d) continue;
          const delta=(C[a][c]+C[b][d])-(C[a][b]+C[c][d]);
          if(delta<-1e-7){ let lo=i,hi=k; while(lo<hi){[seq[lo],seq[hi]]=[seq[hi],seq[lo]];lo++;hi--;} improved=true; }
        }
      }
    }
    return seq;
  }
  function orOpt(seq){ // move chains of length 1..3
    let improved=true;
    while(improved){ improved=false;
      for(let L=1;L<=3;L++){
        for(let i=0;i+L<=seq.length;i++){
          const chain=seq.slice(i,i+L);
          const rest=seq.slice(0,i).concat(seq.slice(i+L));
          const base=routeCost(seq);
          for(let p=0;p<=rest.length;p++){
            if(p===i && L===1) continue;
            const cand=rest.slice(0,p).concat(chain, rest.slice(p));
            if(routeCost(cand)+1e-7<base){ seq.splice(0,seq.length,...cand); improved=true; break; }
          }
          if(improved) break;
        }
        if(improved) break;
      }
    }
    return seq;
  }
  const polish = seq => orOpt(twoOpt(seq));

  // ---- constructions ----
  function kmeans(seed){
    let rng=mulberry(seed);
    const cs=jobs.slice().sort(()=>rng()-0.5).slice(0,K).map(j=>({lat:coords[j].lat,lng:coords[j].lng}));
    let assign=new Array(n).fill(0);
    for(let it=0; it<14; it++){
      for(let a=0;a<n;a++){ const j=jobs[a]; let bi=0,bd=1e18;
        for(let c=0;c<K;c++){const dl=coords[j].lat-cs[c].lat, dg=coords[j].lng-cs[c].lng; const d=dl*dl+dg*dg; if(d<bd){bd=d;bi=c;}}
        assign[a]=bi; }
      const sl=Array.from({length:K},()=>[0,0,0]);
      for(let a=0;a<n;a++){const j=jobs[a];const c=assign[a];sl[c][0]+=coords[j].lat;sl[c][1]+=coords[j].lng;sl[c][2]++;}
      for(let c=0;c<K;c++) if(sl[c][2]){cs[c]={lat:sl[c][0]/sl[c][2],lng:sl[c][1]/sl[c][2]};}
    }
    const routes=Array.from({length:K},()=>[]);
    for(let a=0;a<n;a++) routes[assign[a]].push(jobs[a]);
    return routes;
  }
  function sweep(){
    const ang=jobs.map(j=>({j,a:Math.atan2(coords[j].lat-coords[0].lat, coords[j].lng-coords[0].lng)}))
      .sort((x,y)=>x.a-y.a).map(o=>o.j);
    const routes=Array.from({length:K},()=>[]); const per=Math.ceil(n/K);
    ang.forEach((j,i)=>routes[Math.min(K-1,Math.floor(i/per))].push(j));
    return routes;
  }

  // ---- scoring (depends on balance mode) ----
  // 'finish' : minimize the LONGEST route -> spreads work, every crew gets a route
  // 'count'  : equalize the number of stops per crew
  // 'total'  : minimize total travel (may leave some crews idle)
  function scoreOf(cost, cnts){
    let total=0, mx=0; for(const c of cost){ total+=c; if(c>mx)mx=c; }
    if(balance==='total') return {a:0, b:total};
    if(balance==='finish') return {a:mx, b:total};
    const used=cnts.length; let sum=0; for(const c of cnts) sum+=c; const avg=sum/used;
    let sp=0; for(const c of cnts){ const d=c-avg; sp+=d*d; }
    return {a:sp, b:total};
  }
  const better=(s1,s2)=> (s1.a < s2.a-1e-7) ? true : (s1.a > s2.a+1e-7) ? false : (s1.b < s2.b-1e-6);

  // When the user asks for K crews (and there are >= K jobs), every crew MUST get
  // a route. Only 'total' mode is allowed to leave a crew idle.
  const forceAll = (n>=K); // always use every crew
  function repairEmpties(routes){
    let guard=0;
    while(guard++ < K*3){
      const empties=[]; let donor=-1, mx=1;
      for(let i=0;i<K;i++){ if(routes[i].length===0) empties.push(i); if(routes[i].length>mx){mx=routes[i].length; donor=i;} }
      if(!empties.length || donor<0) break;
      // pull the most outlying stop (farthest from base) out of the biggest route into the empty one
      let bi=0,bv=-1; routes[donor].forEach((job,idx)=>{ const v=C[0][job]; if(v>bv){bv=v;bi=idx;} });
      const job=routes[donor].splice(bi,1)[0];
      routes[empties[0]]=[job];
      if(routes[donor].length) polish(routes[donor]);
    }
    return routes;
  }

  // ---- cross-route local search (relocate + swap), driven by the chosen score ----
  function improve(routes){
    if(forceAll) repairEmpties(routes);
    routes.forEach(r=>{ if(r.length) polish(r); });
    let cost=routes.map(r=> r.length?routeCost(r):0);
    let cnts=routes.map(r=>r.length);
    let cur=scoreOf(cost,cnts);
    let moved=true, guard=0;
    while(moved && guard++<90){
      moved=false;
      // relocate: pull one stop from route a, drop it in the best slot of some route b
      for(let a=0;a<K;a++){
        if(forceAll && routes[a].length<=1) continue; // never empty a required crew
        for(let pi=0; pi<routes[a].length; pi++){
          if(forceAll && routes[a].length<=1) break;  // stop before removing its last stop
          const node=routes[a][pi];
          const without=routes[a].slice(); without.splice(pi,1);
          const ca=without.length?routeCost(without):0;
          let pick=null;
          for(let b=0;b<K;b++){
            if(b===a) continue;
            for(let q=0;q<=routes[b].length;q++){
              const cand=routes[b].slice(); cand.splice(q,0,node);
              const cb=routeCost(cand);
              const nc=cost.slice(); nc[a]=ca; nc[b]=cb;
              const ncn=cnts.slice(); ncn[a]=without.length; ncn[b]=cand.length;
              const s=scoreOf(nc,ncn);
              if(better(s,cur) && (!pick||better(s,pick.s))) pick={b,cand,ca,cb,s,without};
            }
          }
          if(pick){
            routes[a]=pick.without; routes[pick.b]=pick.cand;
            if(routes[a].length)polish(routes[a]); polish(routes[pick.b]);
            cost[a]=routes[a].length?routeCost(routes[a]):0; cost[pick.b]=routeCost(routes[pick.b]);
            cnts[a]=routes[a].length; cnts[pick.b]=routes[pick.b].length;
            cur=scoreOf(cost,cnts); moved=true; pi=-1;
          }
        }
      }
      // swap: trade one stop between two routes
      for(let a=0;a<K;a++) for(let b=a+1;b<K;b++){
        for(let i=0;i<routes[a].length;i++) for(let k=0;k<routes[b].length;k++){
          const ra=routes[a].slice(), rb=routes[b].slice();
          [ra[i],rb[k]]=[rb[k],ra[i]];
          const nc=cost.slice(); nc[a]=routeCost(ra); nc[b]=routeCost(rb);
          const s=scoreOf(nc,cnts); // counts unchanged by a swap
          if(better(s,cur)){
            routes[a]=ra; routes[b]=rb; polish(routes[a]); polish(routes[b]);
            cost[a]=routeCost(routes[a]); cost[b]=routeCost(routes[b]);
            cur=scoreOf(cost,cnts); moved=true;
          }
        }
      }
    }
    const fc=routes.map(r=>r.length?routeCost(r):0), fn=routes.map(r=>r.length);
    return {routes, score:scoreOf(fc,fn)};
  }

  // bias used when rebuilding routes so balance modes don't pile onto one crew
  let avgEdge=0; for(let j=1;j<=n;j++) avgEdge+=C[0][j]; avgEdge/=Math.max(1,n);
  const BALW = balance==='total' ? 0 : avgEdge*0.5;

  // ---- run candidate starts, keep the best by score ----
  let best=null, bestS=null;
  const seeds=[sweep()]; for(let s=0;s<6;s++) seeds.push(kmeans(11+s*7));
  for(const seed of seeds){
    const {routes,score}=improve(seed.map(r=>r.slice()));
    if(!best || better(score,bestS)){ best=routes.map(r=>r.slice()); bestS=score; }
  }
  // ---- ruin & recreate perturbations ----
  let rng=mulberry(98765);
  for(let it=0; it<55; it++){
    let routes=best.map(r=>r.slice());
    const pool=[]; const rm=Math.max(1,Math.round(n*0.2));
    for(let t=0;t<rm;t++){
      const ne=routes.map((r,i)=>r.length?i:-1).filter(i=>i>=0);
      if(!ne.length) break;
      const ri=ne[(rng()*ne.length)|0], pi=(rng()*routes[ri].length)|0;
      pool.push(routes[ri][pi]); routes[ri].splice(pi,1);
    }
    for(const node of pool){
      let bI=0,bP=0,bD=1e18;
      for(let i=0;i<K;i++){ for(let q=0;q<=routes[i].length;q++){
        const before=routes[i].length?routeCost(routes[i]):0;
        const cand=routes[i].slice(); cand.splice(q,0,node);
        const d=(routeCost(cand)-before)+routes[i].length*BALW;
        if(d<bD){bD=d;bI=i;bP=q;}
      }}
      routes[bI].splice(bP,0,node);
    }
    const {routes:imp,score}=improve(routes);
    if(better(score,bestS)){ best=imp.map(r=>r.slice()); bestS=score; }
  }
  best.forEach(r=>{ if(r.length) polish(r); });
  return best;
}

/* ---------------- Scalable solver for big batches (100s of stops) ----------------
   kmeans clustering + nearest-neighbour ordering + capped 2-opt, then delta-based,
   neighbour-limited relocate driven by the same balance score. Runs in well under a
   second for several hundred stops. */
function solveLarge(C, n, K, coords, balance){
  const forceAll=(n>=K); // always use every crew
  const rc=seq=>{let s=0,p=0;for(const v of seq){s+=C[p][v];p=v;}return s+C[p][0];};
  function score(cost,cnts){let total=0,mx=0;for(const c of cost){total+=c;if(c>mx)mx=c;}
    if(balance==='total')return{a:0,b:total};
    if(balance==='finish')return{a:mx,b:total};
    const avg=cnts.reduce((s,c)=>s+c,0)/cnts.length;let sp=0;for(const c of cnts){const d=c-avg;sp+=d*d;}return{a:sp,b:total};}
  const better=(s1,s2)=>(s1.a<s2.a-1e-7)?true:(s1.a>s2.a+1e-7)?false:(s1.b<s2.b-1e-6);
  function kmeans(seed){let rng=mulberry(seed);const idx=[];for(let i=1;i<=n;i++)idx.push(i);
    const cs=idx.slice().sort(()=>rng()-0.5).slice(0,K).map(j=>({lat:coords[j].lat,lng:coords[j].lng}));
    let asg=new Array(n).fill(0);
    for(let it=0;it<12;it++){for(let a=0;a<n;a++){const j=a+1;let bi=0,bd=1e18;for(let c=0;c<K;c++){const dl=coords[j].lat-cs[c].lat,dg=coords[j].lng-cs[c].lng;const d=dl*dl+dg*dg;if(d<bd){bd=d;bi=c;}}asg[a]=bi;}
      const sl=Array.from({length:K},()=>[0,0,0]);for(let a=0;a<n;a++){const j=a+1,c=asg[a];sl[c][0]+=coords[j].lat;sl[c][1]+=coords[j].lng;sl[c][2]++;}
      for(let c=0;c<K;c++)if(sl[c][2]){cs[c]={lat:sl[c][0]/sl[c][2],lng:sl[c][1]/sl[c][2]};}}
    const routes=Array.from({length:K},()=>[]);for(let a=0;a<n;a++)routes[asg[a]].push(a+1);return routes;}
  function nn(arr){const unv=new Set(arr);let cur=0;const out=[];while(unv.size){let best=-1,bd=1e18;for(const j of unv){const d=C[cur][j];if(d<bd){bd=d;best=j;}}out.push(best);unv.delete(best);cur=best;}return out;}
  function twoOpt(seq){const cap=seq.length>250?1:(seq.length>120?3:6);let it=0,imp=true;
    while(imp&&it++<cap){imp=false;for(let i=0;i<seq.length-1;i++){for(let k=i+1;k<seq.length;k++){
      const a=i===0?0:seq[i-1],b=seq[i],c=seq[k],d=k===seq.length-1?0:seq[k+1];if(a===c||b===d)continue;
      if(C[a][c]+C[b][d]<C[a][b]+C[c][d]-1e-9){let lo=i,hi=k;while(lo<hi){const t=seq[lo];seq[lo]=seq[hi];seq[hi]=t;lo++;hi--;}imp=true;}}}}
    return seq;}
  function repair(routes){let g=0;while(g++<K*3){let empties=[],donor=-1,mx=1;for(let i=0;i<K;i++){if(routes[i].length===0)empties.push(i);if(routes[i].length>mx){mx=routes[i].length;donor=i;}}
    if(!empties.length||donor<0)break;let bi=0,bv=-1;routes[donor].forEach((job,idx)=>{const v=C[0][job];if(v>bv){bv=v;bi=idx;}});const job=routes[donor].splice(bi,1)[0];routes[empties[0]]=[job];}return routes;}

  let routes=kmeans(12345).map(r=>nn(r));routes.forEach(r=>{if(r.length>1)twoOpt(r);});
  if(forceAll)repair(routes);
  let cost=routes.map(r=>r.length?rc(r):0), cnts=routes.map(r=>r.length);
  const NB=Math.min(K-1,3), passes=n>300?5:9;
  const centroid=r=>{if(!r.length)return coords[0];let la=0,ln=0;for(const j of r){la+=coords[j].lat;ln+=coords[j].lng;}return{lat:la/r.length,lng:ln/r.length};};
  for(let pass=0;pass<passes;pass++){
    let moved=false;
    const cent=routes.map(centroid);
    const near=routes.map((_,a)=>{const ds=[];for(let b=0;b<K;b++)if(b!==a){const dl=cent[a].lat-cent[b].lat,dg=cent[a].lng-cent[b].lng;ds.push([dl*dl+dg*dg,b]);}ds.sort((x,y)=>x[0]-y[0]);return ds.slice(0,NB).map(x=>x[1]);});
    for(let a=0;a<K;a++){
      for(let pi=0;pi<routes[a].length;pi++){
        if(forceAll&&routes[a].length<=1)break;
        const ra=routes[a], node=ra[pi];
        const prevA=pi===0?0:ra[pi-1], nextA=pi===ra.length-1?0:ra[pi+1];
        const remDelta=C[prevA][nextA]-C[prevA][node]-C[node][nextA];
        const curScore=score(cost,cnts);
        let bestMove=null;
        for(const b of near[a]){
          const rb=routes[b];let bestPos=0,bestIns=1e18;
          for(let pos=0;pos<=rb.length;pos++){const prev=pos===0?0:rb[pos-1],next=pos===rb.length?0:rb[pos];const id=C[prev][node]+C[node][next]-C[prev][next];if(id<bestIns){bestIns=id;bestPos=pos;}}
          const nc=cost.slice();nc[a]=cost[a]+remDelta;nc[b]=cost[b]+bestIns;
          const ncn=cnts.slice();ncn[a]--;ncn[b]++;
          const s=score(nc,ncn);
          if(better(s,curScore)&&(!bestMove||better(s,bestMove.s))){bestMove={b,bestPos,bestIns,s};}
        }
        if(bestMove){const b=bestMove.b;ra.splice(pi,1);routes[b].splice(bestMove.bestPos,0,node);
          cost[a]=ra.length?rc(ra):0;cost[b]=rc(routes[b]);cnts[a]=ra.length;cnts[b]=routes[b].length;moved=true;pi--;}
      }
    }
    if(!moved)break;
  }
  routes.forEach(r=>{if(r.length>1)twoOpt(r);});
  return routes;
}
// Pick the right solver for the problem size.
function solve(C, n, K, coords, balance){ return n>120 ? solveLarge(C,n,K,coords,balance) : solveVRP(C,n,K,coords,balance); }

function mulberry(a){ return function(){ a|=0;a=a+0x6D2B79F5|0; let t=Math.imul(a^a>>>15,1|a); t=t+Math.imul(t^t>>>7,61|t)^t; return ((t^t>>>14)>>>0)/4294967296; }; }

/* ---------------- Plan metrics (shared) ---------------- */
const M2MI=1/1609.344, S2HR=1/3600, FUELSENS=0.5;
function legList(r, dist){ const out=[]; let prev=0; for(const node of r){ out.push(dist[prev][node]*M2MI); prev=node; } return out; }
// Turn solver output (node-id routes) + distance/time matrices into per-route detail and totals.
function buildPlan(routesIdx, dist, dur, P){
  const {depF, retF, rttBuf, rate, gasPrice, mpg}=P;
  const people=Math.max(1, P.people||1);
  const D=Math.max(1, Math.round(P.daysWeek||5)); // depot returns per week
  let totMi=0, totHr=0, maxHr=0, totBuf=0, totDriveHr=0, totFuelMi=0;
  const routes = routesIdx.filter(r=>r.length).map((r)=>{
    let mi=0,driveHr=0,fuelMi=0,prev=0;
    const seq=r.map(node=>node-1);
    for(const node of r){
      const lm=dist[prev][node]*M2MI;
      mi+=lm; driveHr+=dur[prev][node]*S2HR*depF; fuelMi+=lm*(1+(depF-1)*FUELSENS); prev=node;
    }
    const backMi=dist[prev][0]*M2MI, backHr=dur[prev][0]*S2HR*retF;
    mi+=backMi; driveHr+=backHr; fuelMi+=backMi*(1+(retF-1)*FUELSENS);
    const hr=driveHr+rttBuf;
    totMi+=mi; totHr+=hr; totBuf+=rttBuf; totDriveHr+=driveHr; totFuelMi+=fuelMi; maxHr=Math.max(maxHr,hr);
    return { seq, miles:mi, hours:hr, driveHours:driveHr, backMiles:backMi, backHours:backHr, buffer:rttBuf, fuelMiles:fuelMi, legs:legList(r, dist) };
  });
  // ---- Weekly travel with DAILY DEPOT RETURNS (commute-based) ----
  // Each crew drives to its territory and back every working day (the commute repeats
  // D times/week), plus the site-to-site driving once. This makes a closer depot pay
  // off, because the daily commute shrinks. Computed from avg depot distance per crew
  // so it's stable and not distorted by how a single big route happens to be ordered.
  let wkDriveHr=0, wkDriveMi=0, wkFuelMi=0;
  for(const r of routesIdx){
    if(!r.length) continue;
    let depHr=0, depMi=0;
    for(const node of r){ depHr+=dur[0][node]*S2HR; depMi+=dist[0][node]*M2MI; }
    const avgDepHr=depHr/r.length, avgDepMi=depMi/r.length;   // one-way to territory
    let intraHr=0, intraMi=0;
    for(let k=0;k<r.length-1;k++){ intraHr+=dur[r[k]][r[k+1]]*S2HR; intraMi+=dist[r[k]][r[k+1]]*M2MI; }
    wkDriveHr += D*2*avgDepHr + intraHr;                       // D daily round-trip commutes + intra
    wkDriveMi += D*2*avgDepMi + intraMi;
    wkFuelMi  += D*2*avgDepMi + intraMi;
  }
  const labor=totHr*rate*people, gas=(totFuelMi/mpg)*gasPrice, total=labor+gas;
  return { routes, totMi, totHr, maxHr, totBuf, totDriveHr, totFuelMi, labor, gas, total, people,
           wkDriveHr, wkDriveMi, wkFuelMi, daysWeek:D };
}

/* ---------------- Optimize ---------------- */
async function optimize(){
  if(!state.base){ toast('Set a home base first.', false, true); setTimeout(hideToast,2600); return; }
  if(state.jobs.length<1){ toast('Add at least one job stop.', false, true); setTimeout(hideToast,2600); return; }
  let K=Math.max(1, Math.min(500, parseInt($('crews').value)||1));
  K=Math.min(K, state.jobs.length);
  const balance = 'total';   // one consistent goal: minimize total driving, compact territories
  const rate=+$('rate').value||0, gasPrice=+$('gas').value||0, mpg=+$('mpg').value||1;
  const people=Math.max(0.1, parseFloat($('people').value)||1);

  $('run').disabled=true;
  const mat=await getMatrix([state.base, ...state.jobs], m=>toast(m));

  const n=state.jobs.length;
  // matrices indexed 0..n (0 = base)
  const dist=mat.dist, dur=mat.dur;
  const depF=1, retF=1; // no time-of-day traffic
  const C = dist.map(row=>Array.from(row));
  const coords=[ {lat:state.base.lat,lng:state.base.lng}, ...state.jobs.map(j=>({lat:j.lat,lng:j.lng})) ];

  toast(n>120?`Optimizing ${n} stops across ${K} crews…`:'Optimizing crew routes…');
  await new Promise(r=>setTimeout(r,20)); // let UI paint

  const routesIdx = solve(C, n, K, coords, balance); // arrays of node ids (1..n)

  // build report (shared with crew-count comparison)
  const rttBuf = Math.max(0, +$('rttbuf').value||0)/60; // hours added per crew, there-and-back
  const plan = buildPlan(routesIdx, dist, dur, {depF, retF, rttBuf, rate, gasPrice, mpg, people, daysWeek:Math.max(1,parseInt($("daysWeek").value)||5)});
  state.result={ ...plan, matMode:mat.mode, matFactor:mat.factor, matMph:mat.mph };

  drawRoutes();
  drawMarkers();
  renderResult();
  hideToast();
  $('run').disabled=false;
}

/* ---------------- Compare crew counts ---------------- */
async function compareCrews(){
  if(!state.base){ toast('Set a home base first.', false, true); setTimeout(hideToast,2600); return; }
  if(state.jobs.length<2){ toast('Add at least two job stops to compare.', false, true); setTimeout(hideToast,2600); return; }
  const balance = 'total';
  const rate=+$('rate').value||0, gasPrice=+$('gas').value||0, mpg=+$('mpg').value||1;
  const rttBuf=Math.max(0,+$('rttbuf').value||0)/60;
  const people=Math.max(0.1, parseFloat($('people').value)||1);
  const depF=1, retF=1;

  $('compare').disabled=true;
  const mat=await getMatrix([state.base, ...state.jobs], m=>toast(m));

  const n=state.jobs.length, chosenK=Math.max(1,parseInt($('crews').value)||1), maxC=Math.min(n, Math.max(12, chosenK), 40);
  const dist=mat.dist, dur=mat.dur;
  const coords=[ {lat:state.base.lat,lng:state.base.lng}, ...state.jobs.map(j=>({lat:j.lat,lng:j.lng})) ];
  const C = dist.map(r=>Array.from(r));

  const rows=[];
  for(let c=1;c<=maxC;c++){
    toast(`Solving for ${c} crew${c>1?'s':''}…`);
    await new Promise(r=>setTimeout(r,8));
    const ri=solve(C, n, c, coords, balance);
    const p=buildPlan(ri, dist, dur, {depF,retF,rttBuf,rate,gasPrice,mpg,people,daysWeek:Math.max(1,parseInt($("daysWeek").value)||5)});
    rows.push({c, crewsUsed:p.routes.length, totMi:p.totMi, finish:p.maxHr, totHr:p.totHr, labor:p.labor, gas:p.gas, total:p.total});
  }
  hideToast(); $('compare').disabled=false;
  renderCompare(rows);
}

function renderCompare(rows){
  const minCost=Math.min(...rows.map(r=>r.total));
  const minFinish=Math.min(...rows.map(r=>r.finish));
  const fastestC=Math.min(...rows.filter(r=>Math.abs(r.finish-minFinish)<1e-6).map(r=>r.c));
  const maxTotal=Math.max(...rows.map(r=>r.total));
  const body=rows.map(r=>{
    const cheap=r.total===minCost, fast=r.c===fastestC;
    const barW=Math.round(r.total/maxTotal*100);
    const tags=(cheap?'<span class="tg cheap">cheapest</span>':'')+(fast?'<span class="tg fast">fastest</span>':'');
    return `<tr class="${cheap?'r-cheap':''}${fast?' r-fast':''}">
      <td class="cc">${r.c}${r.crewsUsed<r.c?`<i>(${r.crewsUsed} used)</i>`:''}</td>
      <td>${fmtMi(r.totMi)}</td>
      <td>${fmtHrs(r.finish)}</td>
      <td>${fmtHrs(r.totHr)}</td>
      <td>${money(r.labor)}</td>
      <td>${money(r.gas)}</td>
      <td class="tot"><span class="bar" style="width:${barW}%"></span><b>${money(r.total)}</b> ${tags}</td>
    </tr>`;
  }).join('');
  $('compareBody').innerHTML=`
    <table class="cmp">
      <thead><tr><th>Crews</th><th>Total mi</th><th>Finish</th><th>Crew hrs</th><th>Labor</th><th>Gas</th><th>Total cost</th></tr></thead>
      <tbody>${body}</tbody>
    </table>
    <p class="cmpnote">More crews finish the work sooner (lower <b>Finish</b>) but cost more in total, because each crew drives its own round trip from base. Fewer crews are cheaper but slower. Pick the row that fits your deadline and budget, then set that crew count and Optimize.</p>`;
  $('compareModal').classList.add('show');
}

function drawRoutes(){
  lineLayer.clearLayers();
  const many = state.jobs.length>150;
  for(let ri=0; ri<state.result.routes.length; ri++){

    const pts=[ state.base, ...rt.seq.map(i=>state.jobs[i]), state.base ];
    const geo=pts.map(p=>[p.lat,p.lng]); // geometric straight segments in optimized order
    L.polyline(geo,{color:routeColor(ri), weight: many?2:3.2, opacity:.85, lineJoin:'round'}).addTo(lineLayer);
  }
}

/* ---------------- Render results ---------------- */
function renderResult(){
  const R=state.result;
  const rate=+$('rate').value||0, gasPrice=+$('gas').value||0, mpg=+$('mpg').value||1;
  const host=$('tickets'); host.style.display='block'; $('placeholder').style.display='none';
  host.innerHTML='';
  const ppl = R.people||1;

  // ---- Calibration on the commute-based daily-returns weekly model ----
  // Each crew commutes to its territory every working day; a closer depot cuts that
  // commute, which is where a branch split pays off. Factor calibrated so the all-branch
  // route = $3,607/day labor; same factor applies to every run for comparable numbers.
  const daysWeek = Math.max(1, +$('daysWeek').value||5);
  const WEEKS=52;
  const factor = Math.max(0.01, +$('rwFactor').value||1);
  const wkDriveHr=R.wkDriveHr||R.totHr, wkDriveMi=R.wkDriveMi||R.totMi, wkFuelMi=R.wkFuelMi||R.totFuelMi;

  const wkHrs=wkDriveHr*factor,          dayHrs=wkHrs/daysWeek,   yrHrs=wkHrs*WEEKS;
  const wkMiles=wkDriveMi*factor,        dayMiles=wkMiles/daysWeek, yrMiles=wkMiles*WEEKS;
  const wkLabor=wkHrs*ppl*rate,          dayLabor=wkLabor/daysWeek, yrLabor=wkLabor*WEEKS;
  const wkGas=(wkFuelMi*factor/mpg)*gasPrice, dayGas=wkGas/daysWeek, yrGas=wkGas*WEEKS;
  const wkTot=wkLabor+wkGas,             dayTot=dayLabor+dayGas,  yrTot=yrLabor+yrGas;

  // ---- Dashboard: cost split + totals only ----
  $('readout').classList.remove('idle');
  $('mLabor').textContent = money(dayLabor);
  $('mGas').textContent = money(dayGas);
  $('mTotal').textContent = money(dayTot);
  $('mWkTotal').textContent = money(wkTot);
  $('mYrTotal').textContent = money(yrTot);

  // ---- Operating totals card (day / week / year) ----
  const col=(lbl,hrs,lab,gas,tot)=>`
      <div class="ops-col">
        <div class="ops-lbl">${lbl}</div>
        <div class="ops-row"><span>Travel</span><b>${Math.round(hrs).toLocaleString()} h</b></div>
        <div class="ops-row key"><span>Labor</span><b>${money(lab)}</b></div>
        <div class="ops-row"><span>Gas</span><b>${money(gas)}</b></div>
        <div class="ops-row tot"><span>Total</span><b>${money(tot)}</b></div>
      </div>`;
  const ops=document.createElement('div'); ops.className='ops';
  ops.innerHTML = `
    <div class="ops-h">Operating estimate <span>· daily returns ${R.daysWeek||5}×/wk · ×${factor.toFixed(2)} · ${ppl}/crew</span></div>
    <div class="ops-grid">
      ${col('Per day', dayHrs, dayLabor, dayGas, dayTot)}
      ${col('Per week', wkHrs, wkLabor, wkGas, wkTot)}
      ${col('Per year', yrHrs, yrLabor, yrGas, yrTot)}
    </div>
    <div class="ops-foot">Weekly driving with daily depot returns: ${fmtHrs(wkDriveHr)} · ${fmtMi(wkDriveMi)} mi (before ×${factor.toFixed(2)}). Ideal single loop: ${fmtHrs(R.totHr)}.</div>`;
  host.appendChild(ops);

  // data-source context summary
  const tline=document.createElement('div'); tline.className='trafsum';
  tline.innerHTML = `<span class="est">Estimated distances</span> · straight-line ×${(R.matFactor||1).toFixed(2)} for road miles, ~${Math.round(R.matMph||0)} mph`
    + (ppl!==1?` · ${ppl} people/crew`:'');
  host.appendChild(tline);
  const collapse = R.routes.length>0 && R.routes.some(rt=>rt.seq.length>14);
  R.routes.forEach((rt,ri)=>{
    const col=routeColor(ri);
    const rLabor=rt.hours*rate*ppl, rGas=(rt.fuelMiles/mpg)*gasPrice;
    const card=document.createElement('div'); card.className='ticket'; card.style.setProperty('--rc',col);
    let stops;
    if(collapse && rt.seq.length>14){
      const head=rt.seq.slice(0,5).map((idx,pos)=>`<li>${esc(state.jobs[idx].name)}${pos===0?' <em>(drive out)</em>':''}<span class="leg">${fmtMi(rt.legs[pos])} mi</span></li>`).join('');
      const tail=`<li>${esc(state.jobs[rt.seq[rt.seq.length-1]].name)}<span class="leg">${fmtMi(rt.legs[rt.seq.length-1])} mi</span></li>`;
      stops=head+`<li class="more">… ${rt.seq.length-6} more stops …</li>`+tail;
    } else {
      stops=rt.seq.map((idx,pos)=>{
        const tag = pos===0 ? ' <em>(drive out)</em>' : '';
        return `<li>${esc(state.jobs[idx].name)}${tag}<span class="leg">${fmtMi(rt.legs[pos])} mi</span></li>`;
      }).join('');
    }
    const bufLine = rt.buffer>0 ? `<div class="ret">＋ round-trip buffer · ${fmtHrs(rt.buffer)}</div>` : '';
    card.innerHTML=`
      <div class="th"><span class="badge">CREW ${ri+1}</span><span class="cnt">${rt.seq.length} stop${rt.seq.length>1?'s':''}</span></div>
      <div class="stats">
        <div class="st"><span>Miles</span><b>${fmtMi(rt.miles)}</b></div>
        <div class="st"><span>Total time</span><b>${fmtHrs(rt.hours)}</b></div>
        <div class="st"><span>Labor</span><b>${money(rLabor)}</b></div>
        <div class="st"><span>Gas</span><b>${money(rGas)}</b></div>
      </div>
      <ol>${stops}</ol>
      <div class="ret">↩ return to base · ${fmtMi(rt.backMiles)} mi · ${fmtHrs(rt.backHours)}</div>
      ${bufLine}`;
    host.appendChild(card);
  });
}

function parseCoordLine(line){
  // accepts "lat,lng" or "lat lng" (also tolerates a leading label before a comma-free coord pair is ambiguous, so require comma or whitespace-separated numbers only)
  const m=line.match(/^\s*(-?\d{1,3}(?:\.\d+)?)\s*[, ]\s*(-?\d{1,3}(?:\.\d+)?)\s*$/);
  if(!m) return null;
  const la=+m[1], ln=+m[2];
  if(la>=-90&&la<=90&&ln>=-180&&ln<=180) return {lat:+la.toFixed(6), lng:+ln.toFixed(6)};
  return null;
}

function recost(){ // recompute cost + round-trip buffer + headcount without re-routing
  const R=state.result; const rate=+$('rate').value||0, gasPrice=+$('gas').value||0, mpg=+$('mpg').value||1;
  const rttBuf=Math.max(0, +$('rttbuf').value||0)/60;
  const people=Math.max(0.1, parseFloat($('people').value)||1); R.people=people;
  let maxHr=0;
  R.routes.forEach(rt=>{ rt.buffer=rttBuf; rt.hours=rt.driveHours+rttBuf; if(rt.hours>maxHr)maxHr=rt.hours; });
  R.totBuf=rttBuf*R.routes.length;
  R.totHr=R.totDriveHr+R.totBuf;
  R.maxHr=maxHr;
  R.labor=R.totHr*rate*people; R.gas=(R.totFuelMi/mpg)*gasPrice; R.total=R.labor+R.gas; renderResult();
}
